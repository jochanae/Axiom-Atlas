-- ── projects table ────────────────────────────────────────────────────────────
-- (from previous migration — safe no-ops if already applied)

ALTER TABLE projects
  ADD COLUMN IF NOT EXISTS shape jsonb NOT NULL
  DEFAULT '{"identity": [], "constraints": [], "formats": []}'::jsonb;

ALTER TABLE projects ADD COLUMN IF NOT EXISTS node_state jsonb DEFAULT '{}'::jsonb;
ALTER TABLE projects ADD COLUMN IF NOT EXISTS push_history jsonb DEFAULT '[]'::jsonb;
ALTER TABLE projects ADD COLUMN IF NOT EXISTS last_handover_at timestamptz;
ALTER TABLE projects ADD COLUMN IF NOT EXISTS last_handover_hash text;
ALTER TABLE projects ADD COLUMN IF NOT EXISTS last_opened_at timestamptz NOT NULL DEFAULT now();
ALTER TABLE projects ADD COLUMN IF NOT EXISTS entity_type text NOT NULL DEFAULT 'project';


-- ── connections table ─────────────────────────────────────────────────────────
-- These 3 columns exist in the app code but were NEVER added to Neon.
-- Root cause of every dead button related to GitHub / connections:
-- every INSERT into connections was crashing with "column does not exist".

ALTER TABLE connections ADD COLUMN IF NOT EXISTS label    text NOT NULL DEFAULT '';
ALTER TABLE connections ADD COLUMN IF NOT EXISTS url      text;
ALTER TABLE connections ADD COLUMN IF NOT EXISTS metadata jsonb;

-- Back-fill label for existing rows so they read sensibly
UPDATE connections SET label = type WHERE label = '';


-- ── VERIFY ────────────────────────────────────────────────────────────────────
-- Run this after the ALTER statements to confirm it landed on the right DB.
-- You should see rows for: shape, label, url, metadata

SELECT table_name, column_name, data_type
FROM information_schema.columns
WHERE table_name IN ('projects', 'connections')
  AND column_name IN ('shape', 'label', 'url', 'metadata')
ORDER BY table_name, column_name;
